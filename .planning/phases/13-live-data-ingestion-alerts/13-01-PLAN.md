    ```swift
    import Foundation
    import SwiftData
    import OSLog

    // MARK: - DTOs

    struct FuelSADTO: Codable {
        let status: String
        let data: FuelPricesData
    }

    struct FuelPricesData: Codable {
        let petrol95_coastal: Double
        let petrol95_inland: Double
        let diesel50ppm_coastal: Double
        let diesel50ppm_inland: Double
        let last_updated: String

        enum CodingKeys: String, CodingKey {
            case petrol95_coastal = "p95_c"
            case petrol95_inland  = "p95_i"
            case diesel50ppm_coastal = "d50_c"
            case diesel50ppm_inland  = "d50_i"
            case last_updated = "updated_at"
        }
    }

    // MARK: - Errors

    enum FuelSyncError: Error {
        case missingAPIKey
        case invalidResponse
    }

    // MARK: - UserDefaults preference extensions (D-04 to D-07)

    extension UserDefaults {
        /// User's selected pricing region. Defaults to "coastal" (D-05).
        var preferredRegion: String {
            get { string(forKey: "refuel.preferredRegion") ?? "coastal" }
            set { set(newValue, forKey: "refuel.preferredRegion") }
        }

        /// User's selected fuel grade. Defaults to "petrol95" (D-07).
        var preferredGrade: String {
            get { string(forKey: "refuel.preferredGrade") ?? "petrol95" }
            set { set(newValue, forKey: "refuel.preferredGrade") }
        }

        /// Last price value for which a hike alert was sent. Prevents duplicate alerts (D-12).
        /// Returns 0.0 when the key has never been written (fresh install / first sync).
        var lastAlertedPrice: Double {
            get { double(forKey: "refuel.lastAlertedPrice") }
            set { set(newValue, forKey: "refuel.lastAlertedPrice") }
        }
    }

    // MARK: - HikeDetector (pure logic — unit-testable without SwiftData)

    enum HikeDetector {
        /// Returns true when a hike alert should fire (D-01, D-12).
        /// - newPrice: price from the latest API response
        /// - storedPrice: last price value from UserDefaults (0.0 on first run — treated as "not set")
        /// - lastAlertedPrice: last price for which an alert was already sent (duplicate guard)
        ///
        /// First-run guard: if storedPrice is 0.0 the key was never written, which means we have
        /// no baseline to compare against. Any real fuel price would be > 0 and trigger a
        /// false-positive hike alert for every new install. Guard against this explicitly.
        static func shouldAlert(newPrice: Double, storedPrice: Double, lastAlertedPrice: Double) -> Bool {
            guard storedPrice > 0 else { return false }           // first-run guard: no baseline yet
            guard newPrice > storedPrice else { return false }    // D-01: only increases
            guard newPrice != lastAlertedPrice else { return false } // D-12: no duplicate
            return true
        }

        /// Extracts the user-relevant price from a DTO based on grade/region prefs (D-06, D-04).
        static func extractPrice(from dto: FuelSADTO, grade: String, region: String) -> Double {
            switch (grade, region) {
            case ("petrol95", "coastal"): return dto.data.petrol95_coastal
            case ("petrol95", "inland"):  return dto.data.petrol95_inland
            case ("diesel50", "coastal"): return dto.data.diesel50ppm_coastal
            case ("diesel50", "inland"):  return dto.data.diesel50ppm_inland
            default:                      return dto.data.petrol95_coastal
            }
        }
    }

    // MARK: - FuelPriceSyncService

    @ModelActor
    actor FuelPriceSyncService {
        private let baseURL = "https://api.fuelsa.co.za/exapi/fuel/current"
        private let logger = Logger(subsystem: "com.slade.refuel", category: "FuelPriceSyncService")

        // MARK: - Live sync (D-08: API key from bundle, never hardcoded)

        func syncLatestPrices() async throws {
            // D-08: Read key from Info.plist (injected via Secrets.xcconfig build setting)
            guard let apiKey = Bundle.main.infoDictionary?["FUEL_SA_API_KEY"] as? String,
                  !apiKey.isEmpty else {
                logger.error("FUEL_SA_API_KEY not set in Info.plist — skipping live sync")
                throw FuelSyncError.missingAPIKey
            }

            var request = URLRequest(url: URL(string: baseURL)!)
            request.addValue(apiKey, forHTTPHeaderField: "key")
            let (data, _) = try await URLSession.shared.data(for: request)
            let dto = try JSONDecoder().decode(FuelSADTO.self, from: data)

            // Update SwiftData stations, then recalculate analytics
            let fetchDescriptor = FetchDescriptor<Station>()
            let stations = try modelContext.fetch(fetchDescriptor)
            FuelPriceIngestor.calculateAnalytics(for: stations)
            try modelContext.save()

            // D-10: Check for price hike and fire notification if needed
            checkAndAlertIfHike(dto: dto)
        }

        // MARK: - Hike detection and notification (D-01, D-02, D-10, D-12)

        private func checkAndAlertIfHike(dto: FuelSADTO) {
            let grade  = UserDefaults.standard.preferredGrade
            let region = UserDefaults.standard.preferredRegion

            let newPrice       = HikeDetector.extractPrice(from: dto, grade: grade, region: region)
            let storedPrice    = UserDefaults.standard.lastAlertedPrice
            let lastAlertPrice = storedPrice // initial state: stored IS last-alerted baseline

            guard HikeDetector.shouldAlert(newPrice: newPrice,
                                           storedPrice: storedPrice,
                                           lastAlertedPrice: lastAlertPrice) else {
                logger.info("No hike alert needed — newPrice: \(newPrice), storedPrice: \(storedPrice)")
                return
            }

            // Persist before firing so a crash after scheduling still records the alert
            UserDefaults.standard.lastAlertedPrice = newPrice

            // D-02: Alert immediately on detection (no effective_date field in FuelSADTO)
            // NotificationManager is @Observable/@MainActor — dispatch to main actor.
            // scheduleLocalNotification already fails silently when the user has not granted
            // notification permission — no external isAuthorized guard needed here.
            Task { @MainActor in
                let nm = NotificationManager()
                nm.scheduleLocalNotification(
                    title: "Fuel Price Hike Alert",
                    body: "Prices are rising. Fill up now to save!",
                    identifier: "hike_alert_\(Int(newPrice * 100))",
                    timeInterval: 1
                )
            }
        }
    }
    ```

    Implementation notes:
    - HikeDetector is a pure enum (no stored state) — compiler-enforced statelessness, easily unit-tested without SwiftData.
    - First-run guard: `guard storedPrice > 0 else { return false }` prevents a false-positive alert on every fresh install. `UserDefaults.double(forKey:)` returns 0.0 when the key has never been set, which would otherwise make any real fuel price look like a hike. The guard short-circuits before the comparison. On the first actual background sync, storedPrice is 0.0, the guard fires, no alert is sent, and lastAlertedPrice remains unset. On the next sync, if a real price was stored after the first fetch (e.g., by FuelPriceIngestor updating SwiftData), or after the first non-alert sync updates the baseline, the guard passes correctly.
    - The `isAuthorized` guard has been removed from the `Task { @MainActor in }` block. `NotificationManager.scheduleLocalNotification` already calls `UNUserNotificationCenter.add(_:withCompletionHandler:)` which fails silently when not authorised — adding an external `isAuthorized` check introduces a race condition because `isAuthorized` reads the last-cached authorization status, which may not reflect the async `getNotificationSettings` callback result at the moment of the check.
    - UserDefaults extensions live in FuelPriceSyncService.swift to co-locate the preference logic with the service that reads it. If the project grows a dedicated Extensions file, they can be moved.
    - FuelSyncError.missingAPIKey is thrown (not silent) during development so the build remains green while the real key is absent — D-09 compliance.
    - D-11 (silent fail) is handled in the .backgroundTask closure in refuelApp.swift (Plan 03), not here. This service throws and lets the caller handle logging.
