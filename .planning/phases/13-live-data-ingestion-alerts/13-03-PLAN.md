    ```swift
    import SwiftUI
    import SwiftData
    import BackgroundTasks
    import OSLog

    @main
    struct refuelApp: App {
        @Environment(\.scenePhase) private var scenePhase

        static let sharedModelContainer: ModelContainer = {
            let schema = Schema([
                Station.self,
                FuelPrice.self,
                RefuelEvent.self,
                UserProfile.self,
                LuckyDrawEntry.self
            ])

            let modelConfiguration = ModelConfiguration(
                schema: schema,
                isStoredInMemoryOnly: false,
                cloudKitDatabase: .none
            )

            do {
                return try ModelContainer(for: schema, configurations: [modelConfiguration])
            } catch {
                fatalError("Could not create ModelContainer: \(error)")
            }
        }()

        var body: some Scene {
            WindowGroup {
                ContentView(modelContainer: Self.sharedModelContainer)
            }
            .modelContainer(Self.sharedModelContainer)
            // D-10/D-11: Background task handler
            // - Creates FuelPriceSyncService fresh inside the closure (off main actor — correct @ModelActor init)
            // - Silent fail on any error (D-11): logs via OSLog, does NOT surface to user
            // - Always reschedules next refresh (BGAppRefreshTask is one-shot)
            .backgroundTask(.appRefresh("com.slade.refuel.price-sync")) {
                let logger = Logger(subsystem: "com.slade.refuel", category: "BGSync")
                let container = Self.sharedModelContainer
                let syncService = FuelPriceSyncService(modelContainer: container)
                do {
                    try await syncService.syncLatestPrices()
                    logger.info("Background price sync completed successfully")
                } catch {
                    // D-11: silent fail — log only, no user-facing error
                    logger.error("Background price sync failed: \(error.localizedDescription)")
                }
                // Always reschedule — BGAppRefreshTask is one-shot (D-11)
                scheduleNextPriceSync()
            }
            // Schedule on entering background
            .onChange(of: scenePhase) { _, newPhase in
                if newPhase == .background {
                    scheduleNextPriceSync()
                }
            }
        }

        /// Schedules the next BGAppRefreshTask. 15-minute minimum — system exercises discretion on actual delivery.
        private func scheduleNextPriceSync() {
            let request = BGAppRefreshTaskRequest(identifier: "com.slade.refuel.price-sync")
            request.earliestBeginDate = Date(timeIntervalSinceNow: 15 * 60) // 15 minutes
            do {
                try BGTaskScheduler.shared.submit(request)
            } catch {
                Logger(subsystem: "com.slade.refuel", category: "BGSync")
                    .error("Failed to schedule BGAppRefreshTask: \(error.localizedDescription)")
            }
        }
    }
    ```

    Notes:
    - `@Environment(\.scenePhase)` requires no import beyond SwiftUI.
    - `BackgroundTasks` framework import is required for `BGAppRefreshTaskRequest` and `BGTaskScheduler`.
    - `FuelPriceSyncService` is instantiated inside the `.backgroundTask` closure — this is off the main actor by default, which is the correct initialization context for a `@ModelActor` (per RESEARCH.md Pitfall 3).
    - D-09: The identifier "com.slade.refuel.price-sync" must exactly match the value added to BGTaskSchedulerPermittedIdentifiers in Xcode (manual step documented in user_setup frontmatter and in Secrets.xcconfig comments).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Secrets.xcconfig → source control | API key must never cross this boundary — file is gitignored |
| Background task closure → SwiftData | ModelContainer (Sendable) crosses actor boundary safely; ModelContext stays inside actor |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-03-01 | Information Disclosure | Secrets.xcconfig API key in version control | mitigate | File added to .gitignore. Acceptance criteria grep confirms gitignore entry. Build setting FUEL_SA_API_KEY injects via bundle — not in any committed file. |

| T-13-03-03 | Denial of Service | Background task fires too frequently | accept | BGAppRefreshTask minimum 15 minutes; system exercises additional discretion based on battery/network. User cannot force more frequent delivery. Accepted risk — system-managed. |

| T-13-03-05 | Elevation of Privilege | Background fetch capability enables broader background access | accept | Only "Background fetch" mode enabled — not location/VoIP/audio. Scope is minimal and matches the feature requirement (periodic data sync). |
